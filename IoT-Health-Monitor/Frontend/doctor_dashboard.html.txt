<script>
const token = localStorage.getItem('token');

async function fetchPatients(){
  const res = await fetch('http://localhost:5000/api/dashboard/assignedPatients', {
    headers:{'x-auth-token': token}
  });
  const patients = await res.json();
  patients.forEach(p=>loadPatientData(p));
}

async function loadPatientData(patient){
  const res = await fetch(`http://localhost:5000/api/dashboard/patient/${patient._id}`, {
    headers:{'x-auth-token': token}
  });
  const data = await res.json();
  const bpmData = data.feeds.map(f=>f.field1);
  const tempData = data.feeds.map(f=>f.field2);
  const humData = data.feeds.map(f=>f.field3);

  const ctx = document.getElementById(`chart-${patient._id}`).getContext('2d');
  new Chart(ctx,{
    type:'line',
    data:{
      labels: data.feeds.map(f=>f.created_at),
      datasets:[
        {label:'BPM', data:bpmData, borderColor:'red'},
        {label:'Temperature', data:tempData, borderColor:'orange'},
        {label:'Humidity', data:humData, borderColor:'blue'}
      ]
    }
  });

  const alertP = document.querySelector(`#patient-${patient._id} .alert`);
  if(Math.max(...bpmData)>120) alertP.innerText="⚠️ High BPM detected!";
}

function downloadPDF(patientName){
  alert('PDF download feature for '+patientName+' coming soon!');
}

fetchPatients();
</script>
