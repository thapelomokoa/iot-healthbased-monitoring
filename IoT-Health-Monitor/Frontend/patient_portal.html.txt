<script>
const token = localStorage.getItem('token');

async function loadPatientData(){
  const res = await fetch(`http://localhost:5000/api/dashboard/patient/<replace_with_patient_id>`, {
    headers:{'x-auth-token': token}
  });
  const data = await res.json();
  const bpmData = data.feeds.map(f=>f.field1);
  const tempData = data.feeds.map(f=>f.field2);
  const humData = data.feeds.map(f=>f.field3);

  const ctx = document.getElementById('patientChart').getContext('2d');
  new Chart(ctx,{
    type:'line',
    data:{
      labels: data.feeds.map(f=>f.created_at),
      datasets:[
        {label:'BPM', data:bpmData, borderColor:'red'},
        {label:'Temp', data:tempData, borderColor:'orange'},
        {label:'Humidity', data:humData, borderColor:'blue'}
      ]
    }
  });

  if(Math.max(...bpmData)>120) document.querySelector('.alert').innerText="High BPM detected!";
}

function downloadPDF(){
  alert('PDF download coming soon!');
}

loadPatientData();
</script>
